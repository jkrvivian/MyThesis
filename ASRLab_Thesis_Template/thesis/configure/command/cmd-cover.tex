\newcommand{\StartCover}
{
  %
  \singlespacing%
  %
  \StartNewPage
  %
  % 設定使用 無頁碼
  \thispagestyle{empty}
  %
  \EnableCoverPageStyle
  %
  % Aligned to the center of the page
  \begin{center}
} % End of \newcommand{}

\newcommand{\EndCover}
{
  % End of alignment
  \end{center}
  \DisableCoverPageStyle
  \EndOfPage
  \UseDefaultLineStretch
} % End of \newcommand{}
